61 int OptionParser::iniHandler(void *user, char const *section, char const *name, char const *value) {
62  OptionParser *parser = static_cast<OptionParser *>(user);
63  try {
64  BaseOption *option = parser->optionsMap_.at(section).at(name).get();
65  if (option->isDefaulted() || !option->isSet()) {
66  option->parse(parser->options_, value);
67  }
68  } catch (std::out_of_range const &oor) {
69  std::ostringstream message;
70  message << "ERROR: Invalid option in config @file " << section << "." << name << std::endl;
71  throw OptionParsingException(message.str());
72  }
73  return 1;
74 }
