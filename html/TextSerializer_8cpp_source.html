313 void TextSerializer::load(BeliefNode &node, std::istream &is) {
314  std::string line;
315  std::getline(is, line);
316  if (line != "No particles!") {
317  long nParticles;
318  std::string tmpStr;
319  std::istringstream(line) >> nParticles;
320 
321  std::getline(is, line);
322  long numParticlesRead = 0;
323  while (line != "particles end") {
324  std::istringstream sstr(line);
325  for (int i = 0; i < NUM_PARTICLES_PER_LINE; i++) {
326  long seqId, entryId;

328  HistorySequence *sequence = getSolver()->histories_->getSequence(
329  seqId);
330  HistoryEntry *entry = sequence->getEntry(entryId);
331  entry->registerNode(&node);
332  numParticlesRead++;
333  if (numParticlesRead == nParticles) {
334  break;
335  }
336  }
337  std::getline(is, line);
338  }
339  }

341  node.setMapping(loadActionMapping(&node, is));
342  getSolver()->getEstimationStrategy()->setValueEstimator(getSolver(), &node);
343 }
