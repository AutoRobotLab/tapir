75 double BeliefNode::distL1Independent(BeliefNode *b) const {
76  double dist = 0.0;
77  for (HistoryEntry *entry1 : particles_) {
78  for (HistoryEntry *entry2 : b->particles_) {
79  dist += entry1->getState()->distanceTo(*entry2->getState());
80  }
81  }
82  double averageDist = dist / (getNumberOfParticles() * b->getNumberOfParticles());
83  if (averageDist < 0) {
84  debug::show_message("ERROR: Distance < 0 between beliefs.");
85  } else if (averageDist == 0) {
86  debug::show_message("NOTE: Identical belief nodes found!");
87  }
88  return averageDist;
89 }

178 /* -------------------- Core tree-related methods ---------------------- */
179 BeliefNode *BeliefNode::createOrGetChild(Action const &action,
180  Observation const &obs) {
181  ActionNode *actionNode = actionMap_->getActionNode(action);
182  if (actionNode == nullptr) {
183  actionNode = actionMap_->createActionNode(action);
184  actionNode->setMapping(solver_->getObservationPool()->createObservationMapping(actionNode));
185  }
186  BeliefNode *childNode;
187  bool isNew;
188  std::tie(childNode, isNew) = actionNode->createOrGetChild(solver_, obs);
189  if (isNew) {
190  // If we've created a new node, we also need to create some of its key elements.
191  if (data_ != nullptr) {
192  childNode->setHistoricalData(data_->createChild(action, obs));
193  }
194  childNode->setMapping(solver_->getActionPool()->createActionMapping(childNode));
195  solver_->getEstimationStrategy()->setValueEstimator(solver_, childNode);
196  }
197  return childNode;
198 }
