18 void RockSampleMdpSolver::solve() {
19  if (model_->options_->hasVerboseOutput) {
20  std::cout << "Solving MDP...";
21  std::cout.flush();
22  }
23 
24  valueMap_.clear();
25 
26  // States are represented as pairs of integers.
27  // The first number encodes the states of the rocks.
28  // The second number encodes the current position (which rock you're on top of).
29  std::set<std::pair<int, int>> states;
30  std::set<std::pair<int, int>> newStates;
31 
32  // Start with a base case where all rocks are bad.
33  for (int i = 0; i < model_->nRocks_; i++) {
34  newStates.insert(std::make_pair(0, i));
35  }
36 
37  for (int i = 0; i < model_->nRocks_; i++) {
38  states = newStates;
39  newStates.clear();
40  for (std::pair<int, int> entry : states) {
41  long rockStateCode = entry.first;
42  long positionNo = entry.second;
43  GridPosition pos = model_->rockPositions_[positionNo];
44 
45  // The value of going straight to the goal is a simple lower bound.
46  double value = calculateQValue(pos, rockStateCode, -1);
47  if (valueMap_[entry] < value) {
48  valueMap_[entry] = value;
49  } else {
50  value = valueMap_[entry];
51  }
52 
53  for (int j = 0; j < model_->nRocks_; j++) {
54  // Try propagating to the rock we came from - it can't be
55  // the current rock, and we must have sampled it so it
56  // must be bad now.
57  if (j != positionNo && (rockStateCode & (1 << j)) == 0) {
58  long prevCode = rockStateCode | (1 << positionNo);
59  GridPosition prevPos = model_->rockPositions_[j];
60  std::pair<int, int> index = std::make_pair(prevCode, j);
61  double prevValue = calculateQValue(prevPos, prevCode, positionNo);
62  if (valueMap_[index] < prevValue) {
63  valueMap_[index] = prevValue;
64  }
65  newStates.insert(index);
66  }
67  }
68  }
69  }
70 
71  if (model_->options_->hasVerboseOutput) {
72  std::cout << " Done." << std::endl << std::endl;
73  }
74 }

76 double RockSampleMdpSolver::getQValue(RockSampleState const &state) const {
77  GridPosition pos = state.getPosition();
78  long rockStateCode = model_->encodeRocks(state.getRockStates());
79 
80  // Check the value of leaving the map.
81  double value = calculateQValue(pos, rockStateCode, -1);
82 
83  for (int i = 0; i < model_->nRocks_; i++) {
84  // Only try to sample good rocks!
85  if ((rockStateCode & (1 << i)) != 0) {
86  double newValue = calculateQValue(pos, rockStateCode, i);
87  if (newValue > value) {
88  value = newValue;
89  }
90  }
91  }

95 double RockSampleMdpSolver::calculateQValue(GridPosition pos, long rockStateCode,
96  long action) const {
97  long actionsUntilReward = model_->getDistance(pos, action);
98  if (actionsUntilReward == -1) {
99  // Impossible!
100  return -std::numeric_limits<double>::infinity();
101  }
102 
103  double reward;
104  double nextQValue;
105 
106  if (action == -1) {
107  // Reward is received as you move into the goal square, not afterwards.
108  actionsUntilReward -= 1;
109  reward = model_->exitReward_;
110  nextQValue = 0; // Terminal.
111  } else {
112  if ((rockStateCode & (1 << action)) == 0) {
113  debug::show_message("ERROR: No reward for this action!");
114  reward = -model_->badRockPenalty_;
115  } else {
116  reward = model_->goodRockReward_;
117  }
118 
119  long nextCode = rockStateCode & ~(1 << action);
120  nextQValue = valueMap_.at(std::make_pair(nextCode, action));
121  }
122 
123  double discountFactor = model_->options_->discountFactor;
124  return std::pow(discountFactor, actionsUntilReward) * (reward + discountFactor * nextQValue);
125 }
