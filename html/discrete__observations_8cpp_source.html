151 std::unique_ptr<ObservationMapping> DiscreteObservationTextSerializer::loadObservationMapping(
152  ActionNode *owner, std::istream &is) {
153  std::unique_ptr<ObservationMapping> map(
154  getSolver()->getObservationPool()->createObservationMapping(owner));
155  DiscreteObservationMap &discMap =
156  (static_cast<DiscreteObservationMap &>(*map));
157  std::string line;
158  std::getline(is, line);
159  std::string tmpStr;
160  std::istringstream totalsStream(line);
161  long nChildren;
162  totalsStream >> nChildren >> tmpStr >> tmpStr;
163  totalsStream >> discMap.totalVisitCount_;
164 
165  for (int i = 0; i < nChildren; i++) {
166  std::getline(is, line);
167  std::istringstream entryStream(line);
168  std::unique_ptr<Observation> obs = loadObservation(entryStream);
169 
170  entryStream >> tmpStr >> tmpStr;
171  std::getline(entryStream, tmpStr, ';');
172  long childId;
173  std::istringstream(tmpStr) >> childId;
174  long visitCount;
175  entryStream >> visitCount;
176 
177  // Create the mapping entry and set its values.
178  std::unique_ptr<DiscreteObservationMapEntry> entry = (
179  std::make_unique<DiscreteObservationMapEntry>());
180  entry->map_ = &discMap;
181  entry->observation_ = std::move(obs);
182  entry->childNode_ = std::make_unique<BeliefNode>(childId, entry.get(), getSolver());
183  entry->visitCount_ = visitCount;
184 
185  // Add the entry to the map
186  discMap.childMap_.emplace(entry->observation_->copy(), std::move(entry));
187  }
188  // Read the last line for the closing brace.
189  std::getline(is, line);
190  return std::move(map);
191 }
