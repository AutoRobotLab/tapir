58 void MultipleStrategiesExp3::updateStrategyWeights(long strategyNo, double timeUsed,
59  double deltaValue) {
60  StrategyInfo &strategyInfo = strategies_[strategyNo];
61  // Ignore negative changes.
62  if (deltaValue < 0.0) {
63  deltaValue = 0.0;
64  }
65 
66  // Update the parameters for this strategy.
67  strategyInfo.timeSpent += timeUsed;
68  strategyInfo.numberOfTimesUsed++;
69  strategyInfo.weight *= std::exp(
70  strategyExplorationCoefficient_ * (deltaValue / model_->getOptions()->maxVal)
71  / (2 * strategyInfo.probability));
72 
73  // Recalculate the total weight.
74  double weightTotal = 0.0;
75  for (StrategyInfo &info : strategies_) {
76  weightTotal += info.weight;
77  }
78 
79  // Now we calculate actual probabilities for each strategy.
80  double probabilityTotal = 0.0;
81  for (StrategyInfo &info : strategies_) {
82  info.probability = ((1 - strategyExplorationCoefficient_) * info.weight / weightTotal
83  + strategyExplorationCoefficient_ / 2);
84 
85  // Scale the probability by the inverse of the time it costs to use this strategy.
86  info.probability *= (info.numberOfTimesUsed + 1) / (info.timeSpent + 1);
87 
88  probabilityTotal += info.probability;
89  }
90

92  for (StrategyInfo &info : strategies_) {

118  if (status != SearchStatus::UNINITIALIZED) {
