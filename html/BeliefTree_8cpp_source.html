73 void BeliefTree::removeNode(BeliefNode *node) {

75 
76  long lastNodeId = allNodes_.size() - 1;
77 
78  if (id < 0 || id > lastNodeId) {
79  debug::show_message("ERROR: Node ID is out of bounds.");
