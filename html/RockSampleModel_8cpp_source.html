185 void RockSampleModel::recalculateDistances(std::vector<std::vector<int>> &grid,
186  std::vector<GridPosition> targets) {
187  // Preinitialize to -1 for all cells.
188  for (auto &row : grid) {
189  for (auto &cell : row) {
190  cell = -1;
191  }
192  }
193  std::queue<GridPosition> queue;
194  for (GridPosition &pos : targets) {
195  grid[pos.i][pos.j] = 0;
196  queue.push(pos);
197  }
198 
199  while (!queue.empty()) {
200  GridPosition pos = queue.front();
201  queue.pop();
202  int distance = grid[pos.i][pos.j] + 1;
203  for (ActionType direction : {ActionType::NORTH, ActionType::SOUTH, ActionType::WEST,
204  ActionType::EAST}) {
205  GridPosition nextPos;
206  bool isLegal;
207  std::tie(nextPos, isLegal) = makeNextPosition(pos, direction);
208  // The distance matters only if the move is legal, and doesn't move into a goal.
209  if (isLegal && getCellType(nextPos) != GOAL) {
210  int &nextPosDistance = grid[nextPos.i][nextPos.j];
211  if (nextPosDistance == -1 || nextPosDistance > distance) {
212  nextPosDistance = distance;
213  queue.push(nextPos);
214  }
215  }
216  }
217  }
218 }

288 std::pair<GridPosition, bool> RockSampleModel::makeNextPosition(GridPosition position,
289  ActionType actionType) {
290 
291  GridPosition oldPosition = position;
292 
293  bool isLegal = true;
294  if (actionType == ActionType::CHECK) {
295  // Do nothing - the state remains the same.
296  } else if (actionType == ActionType::SAMPLE) {
297  int rockNo = getCellType(position) - ROCK;
298  if (rockNo < 0 || rockNo >= nRocks_) {
299  isLegal = false;
300  }
301  } else if (actionType == ActionType::NORTH) {
302  if (position.i > 0) {
303  position.i -= 1;
304  } else {
305  isLegal = false;
306  }
307  } else if (actionType == ActionType::EAST) {
308  if (position.j < nCols_ - 1) {
309  position.j += 1;
310  } else {
311  isLegal = false;
312  }
313  } else if (actionType == ActionType::SOUTH) {
314  if (position.i < nRows_ - 1) {
315  position.i += 1;
316  } else {
317  isLegal = false;
318  }
319  } else if (actionType == ActionType::WEST) {
320  if (position.j > 0) {
321  position.j -= 1;
322  } else {
323  isLegal = false;
324  }
325  } else {
326  std::ostringstream message;
327  message << "Invalid action: " << actionType;
328  debug::show_message(message.str());
329  isLegal = false;
330  }
331 
332  if (isLegal) {
333  // Moving into obstacles is invalid!
334  if (getCellType(position) == OBSTACLE) {
335  isLegal = false;
336  position = oldPosition;
337  }
338 
339  }
340  return std::make_pair(position, isLegal);
341 }

442 /* -------------- Methods for handling model changes ---------------- */
443 void RockSampleModel::applyChanges(std::vector<std::unique_ptr<solver::ModelChange>> const &changes,
444  solver::Solver *solver) {
445  solver::StatePool *pool = nullptr;
446  if (solver != nullptr) {
447  pool = solver->getStatePool();
448  }
449 
450  if (options_->hasVerboseOutput && pool != nullptr) {
451  cout << "Applying model changes..." << endl;
452  }
453 
454  solver::Heuristic heuristic = getHeuristicFunction();
455  std::vector<double> allHeuristicValues;
456  if (pool != nullptr) {
457  long nStates = pool->getNumberOfStates();
458  allHeuristicValues.resize(nStates);
459  for (long index = 0; index < nStates; index++) {
460  allHeuristicValues[index] = heuristic(nullptr, pool->getInfoById(index)->getState(),
461  nullptr);
462  }
463  }
464 
465  // The cells that have been affected by these changes.
466  std::unordered_set<GridPosition> affectedCells;
467 
468  for (auto const &change : changes) {
469  RockSampleChange const &rsChange = static_cast<RockSampleChange const &>(*change);
470  if (options_->hasVerboseOutput) {
471  cout << rsChange.changeType << " " << rsChange.i0 << " "
472  << rsChange.j0;
473  cout << " " << rsChange.i1 << " " << rsChange.j1 << endl;
474  }
475 
476  RSCellType newCellType;
477  if (rsChange.changeType == "Add Obstacles") {
478  newCellType = OBSTACLE;
479  } else if (rsChange.changeType == "Remove Obstacles") {
480  newCellType = EMPTY;
481  } else {
482  cout << "Invalid change type: " << rsChange.changeType;
483  continue;
484  }
485 
486  for (long i = rsChange.i0; i <= rsChange.i1; i++) {
487  for (long j = rsChange.j0; j <= rsChange.j1; j++) {
488  RSCellType oldCellType = envMap_[i][j];
489  envMap_[i][j] = newCellType;
490  if (newCellType != oldCellType) {
491  affectedCells.insert(GridPosition(i, j));
492  }
493  }
494  }
495 
496  if (pool == nullptr) {
497  continue;
498  }
499 
500  if (searchCategory_ == RSActionCategory::LEGAL && newCellType == RSCellType::EMPTY) {
501  // Legal actions + newly empty cells => handled automatically.
502  }
503 
504  // If we're adding obstacles, we need to mark the invalid states as deleted.
505  solver::RTree *tree = static_cast<solver::RTree *>(pool->getStateIndex());
506 
507  std::vector<double> lowCorner;
508  std::vector<double> highCorner;
509  for (int i = 0; i < nRocks_+2; i++) {
510  lowCorner.push_back(0.0);
511  highCorner.push_back(1.0);
512  }
513  lowCorner[0] = rsChange.i0;
514  lowCorner[1] = rsChange.j0;
515  highCorner[0] = rsChange.i1;
516  highCorner[1] = rsChange.j1;
517 
518 
519  solver::ChangeFlags flags = solver::ChangeFlags::DELETED;
520  if (newCellType == RSCellType::EMPTY) {
521  flags = solver::ChangeFlags::TRANSITION;
522  lowCorner[0] -= 1;
523  lowCorner[1] -= 1;
524  highCorner[0] += 1;
525  highCorner[1] += 1;
526  }
527 
528  solver::FlaggingVisitor visitor(pool, flags);
529  tree->boxQuery(visitor, lowCorner, highCorner);
530  }
531 
532  // Only modify actions if we're working with a legal-only search.
533  if (solver != nullptr && searchCategory_ == RSActionCategory::LEGAL) {
534  LegalActionsPool *actionPool = static_cast<LegalActionsPool *>(solver->getActionPool());
535  for (GridPosition cell : affectedCells) {
536  bool isLegal = (envMap_[cell.i][cell.j] != OBSTACLE);
537  ActionType actionPairs[4][2] = {
538  {ActionType::NORTH, ActionType::SOUTH},
539  {ActionType::EAST, ActionType::WEST},
540  {ActionType::SOUTH, ActionType::NORTH},
541  {ActionType::WEST, ActionType::EAST}
542  };
543 
544  for (auto &pair : actionPairs) {
545  GridPosition adjacentCell = makeAdjacentPosition(cell, pair[0]);
546  // Ignore cells that are out of bounds.
547  if (!isWithinBounds(adjacentCell)) {
548  continue;
549  }
550  actionPool->setLegal(isLegal, adjacentCell, RockSampleAction(pair[1]), solver);
551  }
552  }
553  }
554 
555  // Recalculate all the distances.
556  recalculateAllDistances();
557 
558  if (mdpSolver_ != nullptr) {
559  mdpSolver_->solve();
560  }
561 
562  // Check for heuristic changes.
563  if (pool != nullptr) {
564  long nStates = pool->getNumberOfStates();
565  for (long index = 0; index < nStates; index++) {
566  double oldValue = allHeuristicValues[index];
567  solver::StateInfo *info = pool->getInfoById(index);
568  double newValue = heuristic(nullptr, info->getState(), nullptr);
569  if (std::abs(newValue - oldValue) > 1e-5) {
570  pool->setChangeFlags(info, solver::ChangeFlags::HEURISTIC);
571  }
572  }
573  }
574 
575  if (options_->hasVerboseOutput && pool != nullptr) {
576  cout << "Done applying model changes..." << endl;
577  }
578 }

581 /* ------------ Methods for handling particle depletion -------------- */
582 std::vector<std::unique_ptr<solver::State>> RockSampleModel::generateParticles(
583  solver::BeliefNode */*previousBelief*/, solver::Action const &action,
584  solver::Observation const &obs, long nParticles,
585  std::vector<solver::State const *> const &previousParticles) {
586  std::vector<std::unique_ptr<solver::State>> newParticles;
587 
588  RockSampleAction const &a = static_cast<RockSampleAction const &>(action);
589  if (a.getActionType() == ActionType::CHECK) {
590  long rockNo = a.getRockNo();
591  struct Hash {
592  std::size_t operator()(RockSampleState const &state) const {
593  return state.hash();
594  }
595  };
596  typedef std::unordered_map<RockSampleState, double, Hash> WeightMap;
597  WeightMap weights;
598  double weightTotal = 0;
599  for (solver::State const *state : previousParticles) {
600  RockSampleState const *rockSampleState = static_cast<RockSampleState const *>(state);
601  GridPosition pos(rockSampleState->getPosition());
602  double dist = pos.euclideanDistanceTo(rockPositions_[rockNo]);
603  bool rockIsGood = rockSampleState->getRockStates()[rockNo];
604 
605  double probability = getSensorCorrectnessProbability(dist);
606  RockSampleObservation const &observation =
607  (static_cast<RockSampleObservation const &>(obs));
608  if (rockIsGood != observation.isGood()) {
609  probability = 1 - probability;
610  }
611  weights[*rockSampleState] += probability;
612  weightTotal += probability;
613  }
614  double scale = nParticles / weightTotal;
615  for (WeightMap::value_type &it : weights) {
616  double proportion = it.second * scale;
617  long numToAdd = static_cast<long>(proportion);
618  if (std::bernoulli_distribution(proportion - numToAdd)(*getRandomGenerator())) {
619  numToAdd += 1;
620  }
621  for (int i = 0; i < numToAdd; i++) {
622  newParticles.push_back(std::make_unique<RockSampleState>(it.first));
623  }
624  }
625 
626  } else {
627  // It's not a CHECK action, so we just add each resultant state.
628  for (solver::State const *state : previousParticles) {
629  RockSampleState const *rockSampleState = (static_cast<RockSampleState const *>(state));
630  RockSampleAction const &rockSampleAction =
631  (static_cast<RockSampleAction const &>(action));
632  newParticles.push_back(makeNextState(*rockSampleState, rockSampleAction).first);
633  }
634  }
635  return newParticles;

749 /* ---------------------- Basic customizations ---------------------- */
750 double RockSampleModel::getDefaultHeuristicValue(solver::HistoryEntry const */*entry*/,
751  solver::State const *state, solver::HistoricalData const */*data*/) {
752  RockSampleState const &rockSampleState = static_cast<RockSampleState const &>(*state);
753  double qVal = 0;
754  double currentDiscount = 1;
755  GridPosition currentPos(rockSampleState.getPosition());
756  std::vector<bool> rockStates(rockSampleState.getRockStates());
757 
758  std::set<int> goodRocks;
759  for (int i = 0; i < nRocks_; i++) {
760  // Only bother with reachable rocks.
761  if (rockStates[i] && getDistance(currentPos, i) != -1) {
762  goodRocks.insert(i);
763  }
764  }
765 
766  // Visit the rocks in a greedy order.
767  while (!goodRocks.empty()) {
768  std::set<int>::iterator it = goodRocks.begin();
769  int bestRock = *it;
770  long lowestDist = getDistance(currentPos, bestRock);
771  ++it;
772  for (; it != goodRocks.end(); ++it) {
773  long dist = getDistance(currentPos, *it);
774  if (dist < lowestDist) {
775  bestRock = *it;
776  lowestDist = dist;
777  }
778  }
779  currentDiscount *= std::pow(options_->discountFactor, lowestDist);
780  qVal += currentDiscount * goodRockReward_;
781  goodRocks.erase(bestRock);
782  currentPos = rockPositions_[bestRock];
783  }
784 
785  // Now move to a goal square.
786  currentDiscount *= std::pow(options_->discountFactor, getDistance(currentPos, -1));
787  qVal += currentDiscount * exitReward_;
788  return qVal;
789 }
