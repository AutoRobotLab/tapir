226 bool Solver::isAffected(BeliefNode const *node) {
227  if (changeRoot_ == nullptr) {
228  // No change root => all nodes are affected.
229  return true;
230  }
231  if (node == changeRoot_) {
232  // Same node => must be affected.
233  return true;
234  }
235 
236  long rootDepth = changeRoot_->getDepth();
237  if (node->getDepth() <= rootDepth) {
238  // Not below the root and not the root => can't be affected.
239  return false;
240  }
241 
242  // If it's deeper, look it up in the mapping.
243  std::unordered_map<BeliefNode const *, bool>::iterator it = isAffectedMap_.find(node);
244  if (it != isAffectedMap_.end()) {
245  return it->second;
246  }
247 
248  // We have to traverse the tree to see if it's a descendant.
249  std::vector<BeliefNode const *> ancestors;
250  bool isDescendedFromChangeRoot = false;
251  while (true) {
252  ancestors.push_back(node);
253  node = node->getParentBelief();
254  // If we hit the root, we must be descended from it.
255  if (node == changeRoot_) {
256  isDescendedFromChangeRoot = true;
257  break;
258  }
259  // If we reach the same depth without hitting the root, we aren't descended from it.
260  if (node->getDepth() == rootDepth) {
261  break;
262  }
263 
264  it = isAffectedMap_.find(node);
265  // If we hit a node for which we know whether it's descended, we copy its state.
266  if (it != isAffectedMap_.end()) {
267  isDescendedFromChangeRoot = it->second;
268  break;
269  }
270  }
271 
272  // Add ancestors to the mapping for efficient lookup later on.
273  for (BeliefNode const *ancestor : ancestors) {
274  isAffectedMap_[ancestor] = isDescendedFromChangeRoot;
275  }
276 
277  return isDescendedFromChangeRoot;
278 }

280 void Solver::applyChanges() {
281  std::unordered_set<HistorySequence *> affectedSequences;
282  for (StateInfo *stateInfo : statePool_->getAffectedStates()) {
283  if (changes::has_flags(stateInfo->changeFlags_, ChangeFlags::DELETED)) {
284  // Deletion implies the prior transition must be invalid.
285  stateInfo->changeFlags_ |= ChangeFlags::TRANSITION_BEFORE;
286  }
287 
288  for (HistoryEntry *entry : stateInfo->usedInHistoryEntries_) {
289  BeliefNode *node = entry->getAssociatedBeliefNode();
290 
291  // If this node isn't affected, ignore it.
292  if (!isAffected(node)) {
293  continue;
294  }
295 
296  HistorySequence *sequence = entry->owningSequence_;
297  HistoryEntry::IdType entryId = entry->entryId_;
298 
299  // True iff any changes have been applied to this sequence.
300  bool hasChanged = false;
301  ChangeFlags allFlags = stateInfo->changeFlags_;
302 
303  if (node == changeRoot_) {
304  // Any flags applying to previous states must be ignored.
305  allFlags &= ~ChangeFlags::TRANSITION_BEFORE;
306  allFlags &= ~ChangeFlags::OBSERVATION_BEFORE;
307  allFlags &= ~ChangeFlags::HEURISTIC;
308 
309  // If a state at the change root is deleted, just delete the whole sequence.
310  if (changes::has_flags(allFlags, ChangeFlags::DELETED)) {
311  hasChanged = true;
312  sequence->setChangeFlags(0, ChangeFlags::DELETED);
313  }
314  }
315 
316  // TRANSITION_BEFORE => TRANSITION for previous entry.
317  if (changes::has_flags(allFlags, ChangeFlags::TRANSITION_BEFORE)) {
318  allFlags &= ~ChangeFlags::TRANSITION_BEFORE;
319  if (entryId > 0) {
320  hasChanged = true;
321  sequence->setChangeFlags(entryId - 1, ChangeFlags::TRANSITION);
322  }
323  }
324 
325  // OBSERVATION_BEFORE => OBSERVATION for previous entry.
326  if (changes::has_flags(allFlags, ChangeFlags::OBSERVATION_BEFORE)) {
327  allFlags &= ~ChangeFlags::OBSERVATION_BEFORE;
328  if (entryId > 0) {
329  hasChanged = true;
330  sequence->setChangeFlags(entryId - 1, ChangeFlags::OBSERVATION);
331  }
332  }
333 
334  // Heuristic changes only apply if it's the last entry in the sequence.
335  if(changes::has_flags(allFlags, ChangeFlags::HEURISTIC)) {
336  allFlags &= ~ChangeFlags::HEURISTIC;
337  if (entry->action_ == nullptr) {
338  hasChanged = true;
339  sequence->setChangeFlags(entryId, ChangeFlags::HEURISTIC);
340  }
341  }
342 
343  // Apply any remaining flags, if any.
344  if (allFlags != ChangeFlags::UNCHANGED) {
345  hasChanged = true;
346  sequence->setChangeFlags(entryId, allFlags);
347  }
348  if (hasChanged) {
349  // The sequence is only affected if one of the changes went through.
350  affectedSequences.insert(sequence);
351  }
352  }
353  }
354 
355  long numAffected = affectedSequences.size();
356 
357  if (options_->hasVerboseOutput) {
358  cout << "Must revise " << numAffected << " of ";
359  cout << histories_->getNumberOfSequences() << " histories!" << endl;
360  }
361 
362  // Delete and remove any sequences where the first state has been deleted.
363  std::unordered_set<HistorySequence *>::iterator it = affectedSequences.begin();
364  while (it != affectedSequences.end()) {
365  HistorySequence *sequence = *it;
366  if (changes::has_flags(sequence->getFirstEntry()->changeFlags_, ChangeFlags::DELETED)) {
367  it = affectedSequences.erase(it);
368  // Now we undo the sequence, and delete it entirely.
369  updateSequence(sequence, -1);
370  histories_->deleteSequence(sequence);
371  } else {
372  it++;
373  }
374  }
375 
376  if (options_->hasVerboseOutput) {
377  cout << "Deleted " << numAffected - affectedSequences.size() << " histories!" << endl;
378  }
379 
380  // Revise all of the histories.
381  historyCorrector_->reviseHistories(affectedSequences);
382 
383  if (options_->hasVerboseOutput) {
384  cout << "Revision complete. Backing up..." << endl;
385  }
386 
387  // Finish the deferred backup so that algorithms requiring backed-up values will work properly.
388  doBackup();
389 
390  // Reset the set of affected states in the pool.
391  statePool_->resetAffectedStates();
392 
393  // Now we extend sequences with illegal actions.
394  if (options_->hasVerboseOutput) {
395  cout << "Using search on " << affectedSequences.size() << " histories!" << endl;
396  }
397 
398  // Extend and backup each sequence.
399  for (HistorySequence *sequence : affectedSequences) {
400  searchStrategy_->extendAndBackup(sequence, options_->maximumDepth);
401  }
402 
403  // Backup all the way to the root to keep the tree consistent.
404  doBackup();
405 }

437 void Solver::doBackup() {
438  while (!nodesToBackup_.empty()) {
439  auto firstEntry = nodesToBackup_.cbegin();
440  long depth = firstEntry->first;
441  for (BeliefNode *node : firstEntry->second) {
442  if (depth == 0) {
443  node->recalculateValue();
444  } else {
445  double oldQValue = node->getCachedValue();
446  node->recalculateValue();
447  double deltaQValue = node->getCachedValue() - oldQValue;
448  long nContinuations = node->getMapping()->getTotalVisitCount()
449  - node->getNumberOfStartingSequences();
450  double deltaTotalQ = options_->discountFactor * nContinuations * deltaQValue;
451 
452  ActionMappingEntry *parentActionEntry =
453  node->getParentActionNode()->getParentEntry();
454  if (parentActionEntry->update(0, deltaTotalQ)) {
455  addNodeToBackup(parentActionEntry->getMapping()->getOwner());
456  }
457  }
458  }
459  nodesToBackup_.erase(firstEntry);
460  }
461 }

464 /* ------------------ Methods to update the q-values in the tree. ------------------- */
465 void Solver::updateSequence(HistorySequence *sequence, int sgn, long firstEntryId,
466  bool propagateQChanges) {
467  // Cannot update sequences of length <= 1.
468  if (sequence->getLength() <= 1) {
469  return;
470  }
471 
472  double discountFactor = options_->discountFactor;
473 
474 
475  // Traverse the sequence in reverse.
476  auto it = sequence->entrySequence_.crbegin();
477 
478  // The last entry is used only for the heuristic estimate.
479  double deltaTotalQ = (*it)->immediateReward_;
480  it++;
481  BeliefNode *node;
482  while (true) {
483  // Apply discount and add the immediate reward.
484  deltaTotalQ = deltaTotalQ * discountFactor + (*it)->immediateReward_;
485  node = (*it)->getAssociatedBeliefNode();
486  ActionMapping *mapping = node->getMapping();
487  ActionMappingEntry *entry = mapping->getEntry(*(*it)->getAction());
488  // Update the action value and visit count.
489  entry->update(sgn, sgn * deltaTotalQ);
490 
491  // Update the observation visit count.
492  ObservationMappingEntry *obsEntry = (
493  entry->getActionNode()->getMapping()->getEntry(*(*it)->getObservation()));
494  obsEntry->updateVisitCount(sgn);
495 
496  // If we've gone past the source node, we don't need to update further.
497  // Backpropagation may need to go further, but we can simply defer it.
498  it++;
499  if (it == sequence->entrySequence_.crend() || (*it)->entryId_ < firstEntryId) {
500  addNodeToBackup(node);
501  break;
502  }
503 
504  double oldQ = node->getCachedValue();
505  deltaTotalQ = oldQ;
506  if (propagateQChanges) {
507  // Backpropagate the change in the q-value of the node.
508  node->recalculateValue();
509  double newQ = node->getCachedValue();
510  long nContinuations = mapping->getTotalVisitCount() - node->getNumberOfStartingSequences();
511  deltaTotalQ += (newQ - oldQ) * nContinuations;
512  } else {
513  // If we haven't backpropagated the change, we need to do it later.
514  addNodeToBackup(node);
515  }
516  }
517 }

576 std::function<StateInfo *()> Solver::getStateSampler(BeliefNode *node) {
577  // Nullptr => sample initial sates from the model.
578  if (node == nullptr) {
579  return [this]() {
580  return statePool_->createOrGetInfo(*model_->sampleAnInitState());
581  };
582  }
583 
584  // Filter out any terminal states.
585  std::vector<StateInfo *> nonTerminalStates;
586  for (long index = 0; index < node->getNumberOfParticles(); index++) {
587  HistoryEntry *entry = node->particles_.get(index);
588  if (!model_->isTerminal(*entry->getState())) {
589  nonTerminalStates.push_back(entry->stateInfo_);

591  }
592

596  return std::function<StateInfo *()>();
597  }

600  return [randGen, nonTerminalStates]() {
601  long index = std::uniform_int_distribution<long>(0, nonTerminalStates.size() - 1)(*randGen);
602  return nonTerminalStates[index];
603  };
