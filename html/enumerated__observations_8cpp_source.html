156 EnumeratedObservationTextSerializer::loadObservationMapping(ActionNode *owner,
157  std::istream &is) {
158  std::unique_ptr<ObservationMapping> map(
159 getSolver()->getObservationPool()->createObservationMapping(owner));
160  EnumeratedObservationMap &enumMap = static_cast<EnumeratedObservationMap &>(*map);
161  std::string line;
162  std::getline(is, line);
163  std::string tmpStr;
164  std::istringstream totalsStream(line);
165  long nChildren;
166  totalsStream >> nChildren >> tmpStr >> tmpStr;
167  totalsStream >> enumMap.totalVisitCount_;
168 
169  for (int i = 0; i < nChildren; i++) {
170  std::getline(is, line);
171  std::istringstream entryStream(line);

173 
174  entryStream >> tmpStr >> tmpStr;

176  long childId;

184  entry.childNode_ = std::make_unique<BeliefNode>(childId, &entry, getSolver());
185  enumMap.nChildren_++;
