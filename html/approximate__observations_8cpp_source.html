148 std::unique_ptr<ObservationMapping> ApproximateObservationTextSerializer::loadObservationMapping(
149  ActionNode *owner, std::istream &is) {
150  std::unique_ptr<ObservationMapping> map(
151  getSolver()->getObservationPool()->createObservationMapping(owner));
152  ApproximateObservationMap &approxMap =
153  (static_cast<ApproximateObservationMap &>(*map));
154  std::string line;
155  std::getline(is, line);
156  std::string tmpStr;
157  std::istringstream totalsStream(line);
158  long nChildren;
159  totalsStream >> nChildren >> tmpStr >> tmpStr;
160  totalsStream >> approxMap.totalVisitCount_;
161 
162  for (int i = 0; i < nChildren; i++) {
163  std::getline(is, line);
164  std::istringstream entryStream(line);
165  std::unique_ptr<Observation> obs = loadObservation(entryStream);
166 
167  entryStream >> tmpStr >> tmpStr;
168  std::getline(entryStream, tmpStr, ';');
169  long childId;
170  std::istringstream(tmpStr) >> childId;
171  long visitCount;
172  entryStream >> visitCount;
173 
174  // Create the entry with appropriate values.
175  std::unique_ptr<ApproximateObservationMapEntry> entry = (
176  std::make_unique<ApproximateObservationMapEntry>());
177  entry->map_ = &approxMap;
178  entry->observation_ = std::move(obs);
179  entry->visitCount_ = visitCount;
180  entry->childNode_ = std::make_unique<BeliefNode>(childId, entry.get());
181 
182  // Add the entry to the vector.
183  approxMap.entries_.push_back(std::move(entry));
184  }
185  // Read the last line for the closing brace.
