24 void TagMdpSolver::solve() {
25  if (model_->options_->hasVerboseOutput) {
26  std::cout << "Solving MDP...";
27  std::cout.flush();
28  }
29 
30  valueMap_.clear();
31 
32  // Enumerated vector of actions.
33  std::vector<std::unique_ptr<solver::DiscretizedPoint>> allActions = (
34  model_->getAllActionsInOrder());
35 
36  // Vector of valid grid positions.
37  std::vector<GridPosition> emptyCells;
38  for (long row = 0; row < model_->getNRows(); row++) {
39  for (long col = 0; col < model_->getNCols(); col++) {
40  // Ignore impossible states.
41  if (model_->envMap_[row][col] == TagModel::TagCellType::EMPTY) {
42  emptyCells.emplace_back(row, col);
43  }
44  }
45  }
46 
47  // An initial policy
48  mdp::Policy policy;
49  // Enumerated vector of all states.
50  std::vector<TagState> allStates;
51  // Mapping from each state to its assigned index.
52  std::unordered_map<TagState, int> stateIndex;
53 
54  int index = 0;
55  for (GridPosition const &robotPos : emptyCells) {
56  for (GridPosition const &opponentPos : emptyCells) {
57  TagState state(robotPos, opponentPos, false);
58 
59  ActionType initialAction;
60  // Set a default initial action for our initial policy.
61  if (robotPos == opponentPos) {
62  // Same posn => tag
63  initialAction = ActionType::TAG;
64  } else {
65  // Different posn => try to move towards the opponent.
66  long deltaI = opponentPos.i - robotPos.i;
67  long deltaJ = opponentPos.j - robotPos.j;
68 
69  // Try to move along the axis where we have the greatest difference.
70  if (std::abs(deltaI > std::abs(deltaJ))) {
71  if (deltaI > 0) {
72  initialAction = ActionType::SOUTH;
73  } else {
74  initialAction = ActionType::NORTH;
75  }
76  } else {
77  if (deltaJ > 0) {
78  initialAction = ActionType::EAST;
79  } else {
80  initialAction = ActionType::WEST;
81  }
82  }
83  }
84 
85  allStates.push_back(state);
86  stateIndex[state] = index;
87  policy.push_back(static_cast<int>(initialAction));
88  index += 1;
89  }
90  }
91  // An index of size (past the end of the array!!) will be used to represent terminal states.
92  // A default action for the terminal state is meaningless, but we need it anyway.
93  policy[index] = static_cast<int>(ActionType::TAG);
94 
95 
96  // Initialise the state transitions.
97  std::vector<std::vector<std::unordered_map<int, std::pair<double, double>>>> transitions;
98  transitions.resize(allStates.size() + 1);
99  for (unsigned int stateNo = 0; stateNo <= allStates.size(); stateNo++) {
100  TagState const &state = allStates[stateNo];
101  GridPosition robotPos = state.getRobotPosition();
102  GridPosition opponentPos = state.getOpponentPosition();
103 
104  // Generate a distribution of possible opponent positions.
105  std::unordered_map<GridPosition, double> nextOpponentPosDistribution = (
106  model_->getNextOpponentPositionDistribution(robotPos, opponentPos));;
107 
108  transitions[stateNo].resize(allActions.size());
109  for (unsigned int actionNo = 0; actionNo < allActions.size(); actionNo++) {
110  ActionType actionType = static_cast<ActionType>(actionNo);
111  auto &nextStateTransitions = transitions[stateNo][actionNo];
112 
113  double reward = -1;
114  // The TAG action needs special handling.
115  if (actionType == ActionType::TAG) {
116  if (robotPos == opponentPos) {
117  // Transition to the terminal state
118  nextStateTransitions[allStates.size()] = std::make_pair(1.0, 10.0);
119  continue;
120  } else {
121  // Failure to tag => proceed as usual.
122  reward = -10;
123  }
124  }
125  GridPosition nextRobotPos = model_->getMovedPos(robotPos, actionType).first;
126  for (auto &entry : nextOpponentPosDistribution) {
127  TagState nextState(nextRobotPos, entry.first, false);
128  int nextStateIndex = stateIndex[nextState];
129  nextStateTransitions[nextStateIndex] = std::make_pair(entry.second, reward);
130  }
131  }
132  }
133 
134  std::function<std::vector<int>(int, int)> possibleNextStates =
135  [transitions](int state, int action) {
136  std::vector<int> nextStates;
137  for (auto const &entry : transitions[state][action]) {
138  nextStates.push_back(entry.first);
139  }
140  return std::move(nextStates);
141  };
142  std::function<double(int, int, int)> transitionProbability =
143  [transitions](int state, int action, int nextState) {
144  return transitions[state][action].at(nextState).first;
145  };
146  std::function<double(int, int, int)> reward =
147  [transitions](int state, int action, int nextState) {
148  return transitions[state][action].at(nextState).second;
149  };
150 
151  mdp::PolicyIterator iterator(policy, model_->options_->discountFactor,
152  allStates.size() + 1, allActions.size(),
153  possibleNextStates, transitionProbability, reward);
154 
155  iterator.fixValue(allStates.size(), 0.0);
156 
157  long numSteps = iterator.solve();
158  std::vector<double> stateValues = iterator.getCurrentValues();
159 
160  // Now put all of the state values into our map.
161  for (unsigned int stateNo = 0; stateNo < allStates.size(); stateNo++) {
162  valueMap_[allStates[stateNo]] = stateValues[stateNo];
163  }
164 
165  if (model_->options_->hasVerboseOutput) {
166  std::cout << " Done; took " << numSteps << " steps." << std::endl << std::endl;
167  }
168 }
