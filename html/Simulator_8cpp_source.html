141  if (iter != changeSequence_.end()) {

143  cout << "Model changing." << endl;

155  }

161 
162  if (options_->hasVerboseOutput) {

171  cout << s2 << std::setw(40 - s2.size()) << "";
172  cout << endl;
173  }
174  }

179  return false;
180  }
181  Model::StepResult result = model_->generateStep(*currentState, *action);
182

203  totalReplenishingTime_ += abt::clock_ms() - replenishTimeStart;

219  currentEntry->observation_ = std::move(result.observation);

221  currentEntry->transitionParameters_ = std::move(result.transitionParameters);
222  StateInfo *nextInfo = solver_->getStatePool()->createOrGetInfo(*result.nextState);
223  currentEntry = actualHistory_->addEntry();
224  currentEntry->stateInfo_ = nextInfo;
225 
226  totalDiscountedReward_ += currentDiscount_ * result.reward;
227  currentDiscount_ *= options_->discountFactor;
228  stepCount_++;
229

232  return false;
233  }

238 bool Simulator::handleChanges(std::vector<std::unique_ptr<ModelChange>> const &changes,
239  bool areDynamic) {
240 
241  // Set the change root appropriately.
242  if (areDynamic) {
243  solver_->setChangeRoot(agent_->getCurrentBelief());
244  } else {
245  solver_->setChangeRoot(nullptr);
246  }
247 
248  model_->applyChanges(changes, nullptr);
249 
250  double startTime = abt::clock_ms();
251  solverModel_->applyChanges(changes, solver_);
252  totalChangingTime_ += abt::clock_ms() - startTime;
253 
254  // If the current state is deleted, the simulation is broken!
255  StateInfo const *lastInfo = actualHistory_->getLastEntry()->getStateInfo();
256  if (changes::has_flags(lastInfo->changeFlags_, ChangeFlags::DELETED)) {
257  debug::show_message("ERROR: Current simulation state deleted!");
258  return false;
259  }
260 
261  // If the changes are not dynamic and a past state is deleted, the simulation is broken.
262  if (!areDynamic) {
263  for (HistoryEntry::IdType i = 0; i < actualHistory_->getLength() - 1; i++) {
264  StateInfo const *info = actualHistory_->getEntry(i)->getStateInfo();
265  if (changes::has_flags(info->changeFlags_, ChangeFlags::DELETED)) {
266  std::ostringstream message;
267  message << "ERROR: Impossible simulation history! Includes ";
268  message << *info->getState();
269  debug::show_message(message.str());
270  return false;
271  }
272  }
273  }
274 
275  // Finally we apply the changes.
276  startTime = abt::clock_ms();
277  solver_->applyChanges();
278  totalChangingTime_ += abt::clock_ms() - startTime;
279  return true;
280 }
