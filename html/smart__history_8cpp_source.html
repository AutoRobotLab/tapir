92 std::vector<long> PositionAndRockData::generatePreferredActions() const {
93  std::vector<long> preferredActions;
94 
95  int nRocks = model_->getNumberOfRocks();
96 
97  // Check if we're currently on top of a rock.
98  int rockNo = model_->getCellType(position_) - RockSampleModel::ROCK;
99  // If we are on top of a rock, and it has more +ve than -ve observations
100  // then we will sample it.
101  if (rockNo >= 0 && rockNo < nRocks) {
102  RockData const &rockData = allRockData_[rockNo];
103  if (rockData.chanceGood == 1.0 || rockData.goodnessCount > 0) {
104  preferredActions.push_back(static_cast<long>(ActionType::SAMPLE));
105  return preferredActions;
106  }
107  }
108 
109  bool worthwhileRockFound = false;
110  bool northWorthwhile = false;
111  bool southWorthwhile = false;
112  bool eastWorthwhile = false;
113  bool westWorthwhile = false;
114 
115  // Check to see which rocks are worthwhile.
116  for (int i = 0; i < nRocks; i++) {
117  RockData const &rockData = allRockData_[i];
118  if (rockData.chanceGood != 0.0 && rockData.goodnessCount >= 0) {
119  worthwhileRockFound = true;
120  GridPosition pos = model_->getRockPosition(i);
121  if (pos.i > position_.i) {
122  southWorthwhile = true;
123  } else if (pos.i < position_.i) {
124  northWorthwhile = true;
125  }
126 
127  if (pos.j > position_.j) {
128  eastWorthwhile = true;
129  } else if (pos.j < position_.j) {
130  westWorthwhile = true;
131  }
132  }
133  }
134  // If no rocks are worthwhile head east.
135  if (!worthwhileRockFound) {
136  preferredActions.push_back(static_cast<long>(ActionType::EAST));
137  return preferredActions;
138  }
139 
140  if (northWorthwhile) {
141  preferredActions.push_back(static_cast<long>(ActionType::NORTH));
142  }
143  if (southWorthwhile) {
144  preferredActions.push_back(static_cast<long>(ActionType::SOUTH));
145  }
146  if (eastWorthwhile) {
147  preferredActions.push_back(static_cast<long>(ActionType::EAST));
148  }
149  if (westWorthwhile) {
150  preferredActions.push_back(static_cast<long>(ActionType::WEST));
151  }
152 
153  // See which rocks we might want to check
154  for (int i = 0; i < nRocks; i++) {
155  RockData const &rockData = allRockData_[i];
156  if (rockData.chanceGood != 0.0 && rockData.chanceGood != 1.0
157  && std::abs(rockData.goodnessCount) < 2) {
158  preferredActions.push_back(static_cast<long>(ActionType::CHECK) + i);
159  }
160  }
161  return preferredActions;
162 }
