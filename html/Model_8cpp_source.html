101 std::vector<std::unique_ptr<State>> Model::generateParticles(
102  BeliefNode *previousBelief, Action const &action, Observation const &obs, long nParticles,
103  std::vector<State const *> const &previousParticles) {
104  std::vector<std::unique_ptr<State>> particles;
105  ObservationMapping *obsMap = previousBelief->getMapping()->getActionNode(action)->getMapping();
106  BeliefNode *childNode = obsMap->getBelief(obs);
107 
108  // Use rejection sampling to generate the required number of particles.
109  while ((long)particles.size() < nParticles) {
110  // Sample a random particle.
111  long index = std::uniform_int_distribution<long>(0,
112  previousParticles.size() - 1)(*getRandomGenerator());
113  State const *state = previousParticles[index];
114 
115  // Now generate a step in the model, and compare the observation to the actual observation.
116  // Note that this comparison is done implicitly via the observation mapping, to ensure
117  // that approximate observations are treated cleanly.
118  StepResult result = generateStep(*state, action);
119  if (obsMap->getBelief(*result.observation) == childNode) {
120  particles.push_back(std::move(result.nextState));
121  }
122  }
123  return particles;
124 }
