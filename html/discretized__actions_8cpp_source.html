148 bool DiscretizedActionMapEntry::update(long deltaNVisits, double deltaTotalQ) {
149  if (deltaNVisits == 0 && deltaTotalQ == 0) {
150  return false;
151  }
152 
153  if (!std::isfinite(deltaTotalQ)) {
154  debug::show_message("ERROR: Non-finite delta value!");
155  }
156 
157  if (deltaNVisits > 0 && !isLegal_) {
158  debug::show_message("ERROR: Visiting an illegal action!");
159  }
160 
161  // Update the visit counts
162  if (visitCount_ == 0 && deltaNVisits > 0) {
163  map_->numberOfVisitedEntries_++;
164  // Now that we've tried it at least once, we don't have to try it again.
165  map_->binSequence_.remove(binNumber_);
166  }
167  visitCount_ += deltaNVisits;
168  map_->totalVisitCount_ += deltaNVisits;
169  if (visitCount_ == 0 && deltaNVisits < 0) {
170  map_->numberOfVisitedEntries_--;
171  // Newly unvisited and legal => must try it.
172  if (isLegal_) {
173  map_->binSequence_.add(binNumber_);
174  }
175  }
176 
177  // Update the total Q
178  totalQValue_ += deltaTotalQ;
179 
180  // Update the mean Q
181  double oldMeanQ = meanQValue_;
182  if (visitCount_ <= 0) {
183  meanQValue_ = -std::numeric_limits<double>::infinity();
184  } else {
185  meanQValue_ = totalQValue_ / visitCount_;
186  }
187  return meanQValue_ != oldMeanQ;
188 }
