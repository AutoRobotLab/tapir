24 BeliefNode* NnRolloutFactory::findNeighbor(BeliefNode *belief) {
25  // A maximum distance of 0 means this function is disabled.
26  if (maxNnDistance_ < 0) {
27  return nullptr;
28  }
29 
30  // Initially there is no minimum distance, unless we've already stored a neighbor.
31  double minDist = std::numeric_limits<double>::infinity();
32  BeliefNode *nearestBelief = nnMap_[belief].neighbor;
33  if (nearestBelief != nullptr) {
34  minDist = belief->distL1Independent(nearestBelief);
35  }
36 
37  long numTried = 0;
38  for (BeliefNode *otherBelief : solver_->getPolicy()->getNodes()) {
39  // Obviously we don't want the belief itself.
40  if (belief == otherBelief) {
41  continue;
42  }
43 
44  if (numTried >= maxNnComparisons_) {
45  // Stop if we reach the maximum # of comparisons.
46  break;
47  } else {
48  double distance = belief->distL1Independent(otherBelief);
49  if (distance < minDist) {
50  minDist = distance;
51  nearestBelief = otherBelief;
52  }
53  numTried++;
54  }
55  }
56 
57  // If it's not near enough, we've failed.
58  if (minDist > maxNnDistance_) {
59  return nullptr;
60  }
61 
62  // Otherwise update the mapping with the new neighbor.
63  nnMap_[belief].neighbor = nearestBelief;
64  return nearestBelief;
65 }
