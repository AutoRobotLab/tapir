211 std::unordered_map<GridPosition, double> TagModel::getNextOpponentPositionDistribution(
212  GridPosition const &robotPos, GridPosition const &opponentPos) {
213  std::vector<ActionType> actions = makeOpponentActions(robotPos, opponentPos);

223 GridPosition TagModel::sampleNextOpponentPosition(GridPosition const &robotPos,
224  GridPosition const &opponentPos) {
225  // Randomize to see if the opponent stays still.
226  if (std::bernoulli_distribution(opponentStayProbability_)(
227  *getRandomGenerator())) {
228  return opponentPos;
229  }
230  std::vector<ActionType> actions(makeOpponentActions(robotPos, opponentPos));
231  ActionType action = actions[std::uniform_int_distribution<long>(0,
232  actions.size() - 1)(*getRandomGenerator())];
233  return getMovedPos(opponentPos, action).first;

236 std::pair<GridPosition, bool> TagModel::getMovedPos(GridPosition const &position,
237  ActionType action) {
238  GridPosition movedPos = position;
239  switch (action) {
240  case ActionType::NORTH:
241  movedPos.i -= 1;
242  break;
243  case ActionType::EAST:
244  movedPos.j += 1;
245  break;
246  case ActionType::SOUTH:
247  movedPos.i += 1;
248  break;
249  case ActionType::WEST:
250  movedPos.j -= 1;
251  break;
252  case ActionType::TAG:
253  break;
254  default:
255  std::ostringstream message;
256  message << "Invalid action: " << (long) action;
257  debug::show_message(message.str());
258  break;
259  }
260  bool wasValid = isValid(movedPos);
261  if (!wasValid) {
262  movedPos = position;
263  }
264  return std::make_pair(movedPos, wasValid);

322 /* -------------- Methods for handling model changes ---------------- */
323 void TagModel::applyChanges(std::vector<std::unique_ptr<solver::ModelChange>> const &changes,
324  solver::Solver *solver) {
325  solver::StatePool *pool = nullptr;
326  if (solver != nullptr) {
327  pool = solver->getStatePool();
328  }
329 
330  solver::HeuristicFunction heuristic = getHeuristicFunction();
331  std::vector<double> allHeuristicValues;
332  if (pool != nullptr) {
333  long nStates = pool->getNumberOfStates();
334  allHeuristicValues.resize(nStates);
335  for (long index = 0; index < nStates; index++) {
336  allHeuristicValues[index] = heuristic(nullptr, pool->getInfoById(index)->getState(),
337  nullptr);
338  }
339  }
340 
341  for (auto const &change : changes) {
342  TagChange const &tagChange = static_cast<TagChange const &>(*change);
343  if (options_->hasVerboseOutput) {
344  cout << tagChange.changeType << " " << tagChange.i0 << " "
345  << tagChange.j0;
346  cout << " " << tagChange.i1 << " " << tagChange.j1 << endl;
347  }
348 
349  TagCellType newCellType;
350  if (tagChange.changeType == "Add Obstacles") {
351  newCellType = TagCellType::WALL;
352  } else if (tagChange.changeType == "Remove Obstacles") {
353  newCellType = TagCellType::EMPTY;
354  } else {
355  cout << "Invalid change type: " << tagChange.changeType;
356  continue;
357  }
358 
359  for (long i = tagChange.i0; i <= tagChange.i1; i++) {
360  for (long j = tagChange.j0; j <= tagChange.j1; j++) {
361  envMap_[i][j] = newCellType;
362  }
363  }
364 
365  if (pool == nullptr) {
366  continue;
367  }
368 
369  solver::RTree *tree = static_cast<solver::RTree *>(pool->getStateIndex());
370 
371  double iLo = tagChange.i0;
372  double iHi = tagChange.i1;
373  double iMx = nRows_ - 1.0;
374 
375  double jLo = tagChange.j0;
376  double jHi = tagChange.j1;
377  double jMx = nCols_ - 1.0;
378 
379  // Adding walls => any states where the robot or the opponent are in a wall must
380  // be deleted.
381  if (newCellType == TagCellType::WALL) {
382  solver::FlaggingVisitor visitor(pool, solver::ChangeFlags::DELETED);
383  // Robot is in a wall.
384  tree->boxQuery(visitor,
385  {iLo, jLo, 0.0, 0.0, 0.0},
386  {iHi, jHi, iMx, jMx, 1.0});
387  // Opponent is in a wall.
388  tree->boxQuery(visitor,
389  {0.0, 0.0, iLo, jLo, 0.0},
390  {iMx, jMx, iHi, jHi, 1.0});
391 
392  }
393 
394  // Also, state transitions around the edges of the new / former obstacle must be revised.
395  solver::FlaggingVisitor visitor(pool, solver::ChangeFlags::TRANSITION);
396  tree->boxQuery(visitor,
397  {iLo - 1, jLo - 1, 0.0, 0.0, 0.0},
398  {iHi + 1, jHi + 1, iMx, jMx, 1.0});
399  tree->boxQuery(visitor,
400  {0.0, 0.0, iLo - 1, jLo - 1, 0.0},
401  {iMx, jMx, iHi + 1, jHi + 1, 1.0});
402  }
403 
404  if (mdpSolver_ != nullptr) {
405  mdpSolver_->solve();
406  }
407 
408  // Check for heuristic changes.
409  if (pool != nullptr) {
410  long nStates = pool->getNumberOfStates();
411  for (long index = 0; index < nStates; index++) {
412  double oldValue = allHeuristicValues[index];
413  solver::StateInfo *info = pool->getInfoById(index);
414  double newValue = heuristic(nullptr, info->getState(), nullptr);
415  if (std::abs(newValue - oldValue) > 1e-5) {
416  pool->setChangeFlags(info, solver::ChangeFlags::HEURISTIC);
417  }
418  }
419  }
420 }

423 /* ------------ Methods for handling particle depletion -------------- */
424 std::vector<std::unique_ptr<solver::State>> TagModel::generateParticles(
425  solver::BeliefNode */*previousBelief*/, solver::Action const &action,
426  solver::Observation const &obs,
427  long nParticles,
428  std::vector<solver::State const *> const &previousParticles) {
429  std::vector<std::unique_ptr<solver::State>> newParticles;
430  TagObservation const &observation =
431  (static_cast<TagObservation const &>(obs));
432  ActionType actionType =
433  (static_cast<TagAction const &>(action).getActionType());
434 
435  struct Hash {
436  std::size_t operator()(TagState const &state) const {
437  return state.hash();
438  }
439  };
440  typedef std::unordered_map<TagState, double, Hash> WeightMap;
441  WeightMap weights;
442  double weightTotal = 0;
443 
444  GridPosition newRobotPos(observation.getPosition());
445  if (observation.seesOpponent()) {
446  // If we saw the opponent, we must be in the same place.
447  newParticles.push_back(
448  std::make_unique<TagState>(newRobotPos, newRobotPos,
449  actionType == ActionType::TAG));
450  } else {
451  // We didn't see the opponent, so we must be in different places.
452  for (solver::State const *state : previousParticles) {
453  TagState const *tagState = static_cast<TagState const *>(state);
454  GridPosition oldRobotPos(tagState->getRobotPosition());
455  // Ignore states that do not match knowledge of the robot's position.
456  if (newRobotPos != getMovedPos(oldRobotPos, actionType).first) {
457  continue;
458  }
459 
460  // Get the probability distribution for opponent moves.
461  GridPosition oldOpponentPos(tagState->getOpponentPosition());
462  std::unordered_map<GridPosition, double> opponentPosDistribution = (
463  getNextOpponentPositionDistribution(oldRobotPos, oldOpponentPos));
464 
465  for (auto const &entry : opponentPosDistribution) {
466  if (entry.first != newRobotPos) {
467  TagState newState(newRobotPos, entry.first, false);
468  weights[newState] += entry.second;
469  weightTotal += entry.second;
470  }
471  }
472  }
473  double scale = nParticles / weightTotal;
474  for (WeightMap::iterator it = weights.begin(); it != weights.end();
475  it++) {
476  double proportion = it->second * scale;
477  long numToAdd = static_cast<long>(proportion);
478  if (std::bernoulli_distribution(proportion - numToAdd)(
479  *getRandomGenerator())) {
480  numToAdd += 1;
481  }
482  for (int i = 0; i < numToAdd; i++) {
483  newParticles.push_back(std::make_unique<TagState>(it->first));
484  }
485  }
486  }
487  return newParticles;
488 }

490 std::vector<std::unique_ptr<solver::State>> TagModel::generateParticles(
491  solver::BeliefNode */*previousBelief*/, solver::Action const &action,
492  solver::Observation const &obs, long nParticles) {
493  std::vector<std::unique_ptr<solver::State>> newParticles;
494  TagObservation const &observation =
495  (static_cast<TagObservation const &>(obs));
496  ActionType actionType =
497  (static_cast<TagAction const &>(action).getActionType());
498  GridPosition newRobotPos(observation.getPosition());
499  if (observation.seesOpponent()) {
500  // If we saw the opponent, we must be in the same place.
501  while ((long)newParticles.size() < nParticles) {
502  newParticles.push_back(
503  std::make_unique<TagState>(newRobotPos, newRobotPos,
504  actionType == ActionType::TAG));
505  }
506  } else {
507  while ((long)newParticles.size() < nParticles) {
508  std::unique_ptr<solver::State> state = sampleStateUninformed();
509  solver::Model::StepResult result = generateStep(*state, action);
510  if (obs == *result.observation) {
511  newParticles.push_back(std::move(result.nextState));
512  }
513  }
514  }
515  return newParticles;
516 }

603 /* ---------------------- Basic customizations ---------------------- */
604 double TagModel::getDefaultHeuristicValue(solver::HistoryEntry const */*entry*/,
605  solver::State const *state, solver::HistoricalData const */*data*/) {
606  TagState const &tagState = static_cast<TagState const &>(*state);
607  if (tagState.isTagged()) {
608  return 0;
609  }
610  GridPosition robotPos = tagState.getRobotPosition();
611  GridPosition opponentPos = tagState.getOpponentPosition();
612  long dist = robotPos.manhattanDistanceTo(opponentPos);
613  double nSteps = dist / opponentStayProbability_;
