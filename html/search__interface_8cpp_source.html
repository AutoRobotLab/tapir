81 SearchStatus BasicSearchStrategy::extendAndBackup(HistorySequence *sequence, long maximumDepth) {
82  // We extend from the last entry in the sequence.
83  HistoryEntry *firstEntry = sequence->getLastEntry();
84  long firstEntryId = firstEntry->getId();
85 
86  HistoryEntry *currentEntry = firstEntry;
87  BeliefNode *currentNode = currentEntry->getAssociatedBeliefNode();
88 
89  SearchStatus status = SearchStatus::UNINITIALIZED;
90  std::unique_ptr<StepGenerator> generator = factory_->createGenerator(status, currentEntry,
91  currentEntry->getState(), currentNode->getHistoricalData());
92  if (status == SearchStatus::UNINITIALIZED) {
93  // Failure to initialize => return this status.
94  return status;
95  }
96 
97  // We check for some invalid possibilities in order to deal with them more cleanly.
98  Model *model = solver_->getModel();
99  if (model->isTerminal(*currentEntry->getState())) {
100  debug::show_message("WARNING: Attempted to continue sequence"
101  " from a terminal state.");
102  return SearchStatus::ERROR;
103  } else if (currentEntry->getAction() != nullptr) {
104  debug::show_message("ERROR: The last in the sequence already has an action!?");
105  return SearchStatus::ERROR;
106  } else if (currentEntry->immediateReward_ != 0) {
107  debug::show_message("ERROR: The last in the sequence has a nonzero reward!?");
108  return SearchStatus::ERROR;
109  }
110 
111  while (true) {
112  if (currentNode->getDepth() >= maximumDepth) {
113  // We've hit the depth limit, so we can't generate any more steps in the sequence.
114  status = SearchStatus::OUT_OF_STEPS;
115  break;
116  }
117  // Step the search forward.
118  Model::StepResult result = generator->getStep(currentEntry, currentEntry->getState(),
119  currentNode->getHistoricalData());
120 
121  // Null action => stop the search.
122  if (result.action == nullptr) {
123  break;
124  }
125 
126  // Set the parameters of the current history entry using the ones we got from the result.
127  currentEntry->immediateReward_ = result.reward;
128  currentEntry->action_ = std::move(result.action);
129  currentEntry->transitionParameters_ = std::move(result.transitionParameters);
130  currentEntry->observation_ = std::move(result.observation);
131 
132  // Create the child belief node, and set the current node to be that node.
133  BeliefNode *nextNode = currentNode->createOrGetChild(*currentEntry->action_,
134  *currentEntry->observation_);
135  currentNode = nextNode;
136 
137  // Now we create a new history entry and step the history forward.
138  StateInfo *nextStateInfo = solver_->getStatePool()->createOrGetInfo(*result.nextState);
139  currentEntry = sequence->addEntry();
140 
141  // Register the new history entry with its state, and with its associated belief node.
142  currentEntry->registerState(nextStateInfo);
143  currentEntry->registerNode(currentNode);
144 
145  if (result.isTerminal) {
146  // Terminal state => search complete.
147  status = SearchStatus::FINISHED;
148  break;
149  }
150  }
151 
152  // OUT_OF_STEPS => must calculated a heuristic estimate.
153  if (status == SearchStatus::OUT_OF_STEPS) {
154  currentEntry->immediateReward_ = heuristic_(currentEntry, currentEntry->getState(),
155  currentNode->getHistoricalData());
156  status = SearchStatus::FINISHED;
157  }
158 
159  if (status == SearchStatus::FINISHED) {
160  // Now that we're finished, we back up the sequence.
161  if (firstEntryId > 0 && currentEntry->getId() > firstEntryId) {
162  // If we've extended a previously terminating sequence, we have to add a continuation.
163  solver_->updateEstimate(firstEntry->getAssociatedBeliefNode(), 0, +1);
164  }
165  // We only do a partial backup along the newly generated part of the sequence.
166  solver_->updateSequence(sequence, +1, firstEntryId);
167  } else {
168  // This shouldn't happen => print out an error message.
169  if (status == SearchStatus::UNINITIALIZED) {
170  debug::show_message("ERROR: Search algorithm could not initialize.");
171  } else if (status == SearchStatus::INITIAL) {
172  debug::show_message("ERROR: Search algorithm initialized but did not run!?");
173  } else if (status == SearchStatus::ERROR) {
174  debug::show_message("ERROR: Error in search algorithm!");
175  } else {
176  debug::show_message("ERROR: Invalid search status.");
177  }
178  }
179 
180  // Finally, we just return the status.
181  return status;
182 }
